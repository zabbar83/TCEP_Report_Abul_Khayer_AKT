SELECT             v.teritory_name AS territoryName,
                                                           v.zone_name AS zoneName,
                                                           v.route_name AS routeName,
                                                           c.TRADE_LICENSE_NAME AS outletName,
                                                           c.CUSTOMER_NAME AS outletOwner,
                                                           c.MOBILE_NO AS outletMobile,
                                                           ba.br_team_team_id AS teamId,
                                                           bt.team_name AS teamName,
                                                           bu.user_name AS brUserName,
                                                           brep.name AS brName,
                                                           CASE
                                                               WHEN brep.br_designation = 3 THEN 'BR'
                                                           END AS brDesignation,
                                                           brep.contact_number AS brMobile,
                                                           ba.add_date_time AS addDate,
                                                           bc.customer_name AS customerName,
                                                           bc.age AS customerAge,
                                                           bcon.contact_number AS customerMobile,
                                                           bc.district AS district,
                                                           bc.thana AS thana,
                                                           bc.id AS brActivityConsumerId,
                                                           bc.customer_address AS customerAddress,
                                                           ba.is_repeat_contact AS isRepeatContact,
                                                           ba.ID AS brActivityId,
                                                           ba.consumer_type as consumerType,
                                                           ba.CLUSTER_NAME as clusterName,
                                                           TO_CHAR(ba.ACTIVITY_START_TIME) as activityStartTime,
                                                           TO_CHAR(ba.ACTIVITY_END_TIME) as activityEndTime,
                                                           LPAD(EXTRACT(HOUR FROM (ba.ACTIVITY_END_TIME-ba.ACTIVITY_START_TIME)), 2, '0') || ':' ||
                                                           LPAD(EXTRACT(MINUTE FROM (ba.ACTIVITY_END_TIME-ba.ACTIVITY_START_TIME)), 2, '0') || ':' ||
                                                           LPAD(TRUNC(EXTRACT(SECOND FROM (ba.ACTIVITY_END_TIME-ba.ACTIVITY_START_TIME))), 2, '0') ||
                                                           '.' ||
                                                           LPAD(TRUNC(MOD(EXTRACT(SECOND FROM (ba.ACTIVITY_END_TIME-ba.ACTIVITY_START_TIME)),1)*1000), 3, '0') as duration,
                                                           NVL(CASE
                                                               WHEN ba.using_brand = 19 THEN 'ESSE_SLIM'
                                                               WHEN ba.using_brand = 20 THEN 'MOND_SLIM'
                                                               WHEN ba.using_brand = 21 THEN 'ORIS_SLIM'
                                                               WHEN ba.using_brand = 22 THEN 'XSO_SLIM'
                                                               WHEN ba.using_brand = 23 THEN 'ORIS_NANO'
                                                               WHEN ba.using_brand = 24 THEN 'PATRON_NANO'
                                                               WHEN ba.using_brand = 17 THEN 'CAMEL'
                                                               WHEN ba.using_brand = 25 THEN 'JPGL'
                                                               WHEN ba.using_brand = 26 THEN 'B_H_L'
                                                               WHEN ba.using_brand = 27 THEN 'B_H_S'
                                                               WHEN ba.using_brand = 28 THEN 'MARLBORO'
                                                               WHEN ba.using_brand = 12 THEN 'LUCKY_STRIKE'
                                                           END,'OTHERS') AS usingBrand,
                                                           p.product_name AS offeredBrand,
                                                           p.product_id AS offeredBrandProductId,
                                                           ba.contact_module AS contactModule,
                                                           CASE
                                                               WHEN ba.swapping_no >0 then 'Yes'
                                                               ELSE 'No'
                                                           END AS swappingStatus,
                                                           CASE
                                                               WHEN ba.swapping_no =1 then 1
                                                               ELSE 0
                                                           END AS swapping1,
                                                           CASE
                                                               WHEN ba.swapping_no =6 then 1
                                                               ELSE 0
                                                           END AS swapping6,
                                                           CASE
                                                               WHEN vs.quantity >0 then 'Yes'
                                                               ELSE 'No'
                                                           END AS vaoStatus,
                                                           CASE
                                                               WHEN vs.product_quantity =5 then 1
                                                               ELSE 0
                                                           END AS vao5s,
                                                           CASE
                                                               WHEN vs.product_quantity =20 then 1
                                                               ELSE 0
                                                           END AS vao20s,
                                                           CASE
                                                               WHEN vs.product_quantity =40 then 1
                                                               ELSE 0
                                                           END AS vao40s,                                                           
                                                           CASE
                                                               WHEN ba.sampling_no >0 then 'Yes'
                                                               ELSE 'No'
                                                           END AS samplingStatus,
                                                           ba.sampling_no
                                                    FROM appce.bractivity ba
                                                    JOIN appce.v_teritory_all v on ba.route_id=v.route_id
                                                    and     ( :divisionId is null or v.division_id = :divisionId)
                                                   and     ( :teritoryId is null or v.teritory_id = :teritoryId)
                                                    AND  ba.target_date =to_date( :targetDate,'DD-MON-YY')
                                                    JOIN appce.bruser bu ON ba.br_user_id = bu.user_id
                                                    JOIN appce.brand_representative brep ON ba.br_user_id = brep.br_user_user_id
                                                    JOIN appce.brteam bt ON ba.br_team_team_id = bt.team_id
                                                    JOIN appce.consumer c ON c.id = ba.outlet_id
                                                    JOIN appce.bractivity_consumer bc ON ba.br_activity_consumer_id = bc.id
                                                    JOIN appce.bractivity_contact bcon ON bc.br_activity_contact_id = bcon.id
                                                    JOIN appce.product p ON ba.offered_brand = p.product_id
                                                    AND  p.product_desc='ARIS'
                                                    LEFT JOIN appce.vao_sale vs ON vs.bractivity_id = ba.id
                                                    ORDER BY ba.id DESC
