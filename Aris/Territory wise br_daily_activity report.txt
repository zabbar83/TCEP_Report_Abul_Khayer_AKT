select  DISTINCT
                            vta.WING_ID wingId,
                            vta.WING_NAME wingName,
                            vta.DIVISION_ID divisionId,
                            vta.DIVISION_NAME divisionName,
                            vta.TERITORY_ID teritoryId,
                            vta.TERITORY_NAME teritoryName,
                            pa.allocation allocation,
                            NVL(bat.user_attendance,0) userAttendance,
                            NVL(bat.user_attendance,0) * 20 contactTarget,
                            NVL(orginal,0) orginal,
                            NVL(apple,0) apple,
                            NVL(total_con,0) totalCon,
                            ROUND(case when (NVL(bat.user_attendance,0) * 20) > 0 then (NVL(total_con,0) / (NVL(bat.user_attendance,0) * 20) ) * 100 else 0 end,2) totalConAch,
                            
                            NVL(bat.user_attendance,0) * 20 swappingTarget,
                            NVL(swappingCon1,0) swappingCon1,
                            NVL(swappingCon6,0) swappingCon6,
                            NVL(swappingCon,0) swappingCon,
                            ROUND(case when (NVL(bat.user_attendance,0) * 20) > 0 then (NVL(swappingCon,0) / (NVL(bat.user_attendance,0) * 20) ) * 100 else 0 end,2) swappingAch,
                            
                            NVL(bat.user_attendance,0) * 6 vao5sTarget,
                            NVL(vao5s,0) vao5s,
                            ROUND(case when (NVL(bat.user_attendance,0) * 6) > 0 then (NVL(vao5s,0) / (NVL(bat.user_attendance,0) * 6) ) * 100 else 0 end,2) vao5sAch,
                            
                            NVL(bat.user_attendance,0) * 3 vao20sTarget,
                            NVL(vao20s,0) vao20s,
                            ROUND(case when (NVL(bat.user_attendance,0) * 3) > 0 then (NVL(vao20s,0) / (NVL(bat.user_attendance,0) * 3) ) * 100 else 0 end,2) vao20sAch,
                            
                            NVL(bat.user_attendance,0) * 1 vao40sTarget,
                            NVL(vao40s,0) vao40s,
                            ROUND(case when (NVL(bat.user_attendance,0) * 1) > 0 then (NVL(vao40s,0) / (NVL(bat.user_attendance,0) * 1) ) * 100 else 0 end,2) vao40sAch,
                            
                            (NVL(bat.user_attendance,0) * 6) + (NVL(bat.user_attendance,0) * 3) + (NVL(bat.user_attendance,0) * 1) vaoTotalTarget,
                            NVL(vaoTotal,0) vaoTotal,
                            ROUND(case when ((NVL(bat.user_attendance,0) * 6) + (NVL(bat.user_attendance,0) * 3) + (NVL(bat.user_attendance,0) * 1)) > 0 then (NVL(vaoTotal,0) / ((NVL(bat.user_attendance,0) * 6) + (NVL(bat.user_attendance,0) * 3) + (NVL(bat.user_attendance,0) * 1)) ) * 100 else 0 end,2) vaoTotalAch
                    FROM    appce.ASSIGN_POSITION_ALLOCATION apa 
                    JOIN    appce.POSITION_ALLOCATION_POLICY pap ON apa.POSITION_ALLOCATION_POLICY_ID = pap.ID  
                    and     apa.TARGET_DATE= to_date( :targetDate, 'DD-MON-YY')
                    JOIN    appce.POSITION_ALLOCATION pa ON (pa.POSITION_ALLOCATION_POLICY_ID = pap.ID and pa.role_id=14)
                    JOIN    appce.V_TERITORY_ALL vta ON apa.TERITORY_ID = vta.TERITORY_ID
                    and     ( :divisionId is null or vta.division_id = :divisionId)
                   and     ( :teritoryId is null or vta.teritory_id = :teritoryId)
                    LEFT    JOIN (SELECT  bu.TERITORY_ID,count(bua.BR_USER_USER_ID) user_attendance
                                FROM    appce. bruser bu 
                                JOIN    appce. bruser_attendance bua ON bua.BR_USER_USER_ID = bu.USER_ID 
                                AND     to_date(TO_CHAR(bua.ATTENDANCE_DATE_TIME, 'DD-MON-YY'), 'DD-MON-YY')= to_date( :targetDate, 'DD-MON-YY')
                                JOIN    appce. bruser_roles bur ON (bur.BRUSER_USER_ID = bu.USER_ID and bur.ROLES_ID = 14)
                                WHERE   bu.IS_ACCOUNT_NON_EXPIRED = 1
                                GROUP   BY bu.TERITORY_ID)bat
                                ON apa.TERITORY_ID = bat.TERITORY_ID
                    LEFT    JOIN (select  vt.teritory_id,
                                    sum(orginal) orginal,
                                    sum(apple) apple,
                                    sum(total_con) total_con,
                                    sum(swappingCon1) swappingCon1,
                                    sum(swappingCon6) swappingCon6,
                                    sum(swappingCon) swappingCon,
                                    sum(vao5s) vao5s,
                                    sum(vao20s) vao20s,
                                    sum(vao40s) vao40s,
                                    sum(vaoTotal) vaoTotal
                            from
                            (select  ba.route_id,
                                    count(distinct (case when ba.offered_brand = 32 then ba.id else null end)) orginal,
                                    count(distinct (case when ba.offered_brand = 33 then ba.id else null end)) apple,
                                    count(distinct (case when ba.offered_brand IN (32,33) then ba.id else null end)) total_con,
                                    sum(case when ba.offered_brand IN (32,33) and swapping_no =1 then 1 else 0 end) swappingCon1,
                                    sum(case when ba.offered_brand IN (32,33) and swapping_no =6 then 1 else 0 end) swappingCon6,
                                    sum(case when ba.offered_brand IN (32,33) and swapping_no > 0 then 1 else 0 end) swappingCon,
                                    sum(case when vs.product_quantity = 5 then vs.quantity else 0 end) vao5s,
                                    sum(case when vs.product_quantity = 20 then vs.quantity else 0 end) vao20s,
                                    sum(case when vs.product_quantity = 40 then vs.quantity else 0 end) vao40s,
                                    sum(case when vs.product_quantity IN(5,20,40) then vs.quantity else 0 end) vaoTotal
                            from    ( select * from appce. bractivity where TARGET_DATE =to_date( :targetDate, 'DD-MON-YY')) ba
                            LEFT    JOIN appce.vao_sale vs on ba.id=vs.bractivity_id
                            group   by ba.route_id
                            ) bac
                            JOIN   appce. v_teritory_all vt on bac.route_id=vt.route_id
                            and     ( :divisionId is null or vt.division_id = :divisionId)
                            and     ( :teritoryId is null or vt.teritory_id = :teritoryId)
                            group   by vt.teritory_id
                            )bra
                            ON apa.TERITORY_ID = bra.TERITORY_ID
                    ORDER  BY vta.WING_ID,vta.DIVISION_ID,vta.TERITORY_ID
